CREATE DATABASE HTTTKT
GO

USE HTTTKT
GO

DROP TABLE IF EXISTS GOC_PHIEU;

CREATE TABLE GOC_PHIEU
(
	PSO VARCHAR(20) PRIMARY KEY,
	MSCH INT,
	NLAP DATE,
	MSCH1 INT,
	MSKH INT,
	SOHD VARCHAR(50),
	LOAIHD VARCHAR(50),
	NGPHHD DATE,
	NTTOAN DATE,
	QUYEN_HD INT,
	LYDO NVARCHAR(200),
	CTGOC VARCHAR(50),
	NAM_THANG DATE
);

DROP TABLE IF EXISTS HANG_HOA;

CREATE TABLE HANG_HOA
(
	PSO VARCHAR(20),
	MAHH INT,
	DVT VARCHAR(20),
	DONGIA FLOAT(8),
	THUE_SUAT FLOAT(8),
	TIEN_THUE FLOAT(8),
	PRIMARY KEY(PSO,MAHH)
)

DROP TABLE IF EXISTS DANHMUC_HANGHOA;

CREATE TABLE DANHMUC_HANGHOA
(
	MAHH INT PRIMARY KEY IDENTITY,
	TENHANGHOA NVARCHAR(50),
	DVT VARCHAR(20)
)

DROP TABLE IF EXISTS BAOCAONHAPXUATTON;

CREATE TABLE BAOCAONHAPXUATTON
(
	THANG DATE,
	MSCH INT,
	MAHH INT,
	SL_NHAP INT,
	SL_XUAT INT,
	SL_TON INT,
	PRIMARY KEY(THANG,MSCH,MAHH)
	
)

DROP TABLE IF EXISTS GIAVON_BQGQ;

CREATE TABLE GIAVON_BQGQ
(
	THANG DATE,
	MSCH INT,
	MAHH INT,
	GIAVON_BQGQ FLOAT(8),
	PRIMARY KEY(THANG,MSCH,MAHH)
)
